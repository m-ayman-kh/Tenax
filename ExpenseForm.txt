<script>
  // Parse config from server-side template variable
  var CONFIG = JSON.parse('<?= configJson ?>');
  
  (function() {
    var today = new Date().toISOString().split('T')[0];
    document.getElementById('date').setAttribute('max', today);
    document.getElementById('date').value = today;
    
    // Populate categories
    var categorySelect = document.getElementById('category');
    var categories = CONFIG.EXPENSE_CATEGORIES;
    
    for (var cat in categories) {
      if (categories.hasOwnProperty(cat)) {
        var option = document.createElement('option');
        option.value = cat;
        option.textContent = cat;
        categorySelect.appendChild(option);
      }
    }
    
    // Populate months
    var monthGrid = document.getElementById('monthGrid');
    var months = CONFIG.MONTHS;
    
    for (var i = 0; i < months.length; i++) {
      var month = months[i];
      var btn = document.createElement('div');
      btn.className = 'month-btn';
      btn.dataset.month = month;
      // Convert yy-mmm to mmm-yy for display
      var parts = month.split('-');
      btn.textContent = parts[1] + '-' + parts[0];
      btn.onclick = function() { toggleMonth(this); };
      monthGrid.appendChild(btn);
    }
    
    var selectedMonths = [];
    var selectedFiles = [];
    var MAX_FILES = CONFIG.MAX_ATTACHMENTS;
    var MAX_SIZE = CONFIG.MAX_FILE_SIZE_MB * 1024 * 1024;
    
    window.updateSubCategories = function() {
      var category = document.getElementById('category').value;
      var subSelect = document.getElementById('subCategory');
      
      subSelect.innerHTML = '<option value="">Select Sub-Category</option>';
      
      if (category && CONFIG.EXPENSE_CATEGORIES[category]) {
        subSelect.disabled = false;
        var subs = CONFIG.EXPENSE_CATEGORIES[category];
        for (var i = 0; i < subs.length; i++) {
          var option = document.createElement('option');
          option.value = subs[i];
          option.textContent = subs[i];
          subSelect.appendChild(option);
        }
      } else {
        subSelect.disabled = true;
      }
    };
    
    window.toggleMonth = function(element) {
      var month = element.dataset.month;
      
      if (element.classList.contains('selected')) {
        element.classList.remove('selected');
        selectedMonths = selectedMonths.filter(function(m) { return m !== month; });
      } else {
        element.classList.add('selected');
        selectedMonths.push(month);
      }
      
      document.getElementById('monthSummary').textContent = selectedMonths.length + ' month(s) selected';
    };
    
    window.validateAmount = function(input) {
      if (input.value < 0) {
        input.value = Math.abs(input.value);
      }
      
      var error = document.getElementById('amountError');
      if (input.value && parseFloat(input.value) <= 0) {
        error.style.display = 'block';
        input.style.borderColor = '#f5576c';
      } else {
        error.style.display = 'none';
        input.style.borderColor = '#e0e0e0';
      }
    };
    
    window.handleFiles = function(files) {
      Array.from(files).forEach(function(file) {
        if (selectedFiles.length >= MAX_FILES) {
          alert('Maximum ' + MAX_FILES + ' files allowed');
          return;
        }
        
        if (file.size > MAX_SIZE) {
          alert(file.name + ' is too large. Max size: ' + CONFIG.MAX_FILE_SIZE_MB + 'MB');
          return;
        }
        
        selectedFiles.push(file);
        renderFileList();
      });
    };
    
    function renderFileList() {
      var fileList = document.getElementById('fileList');
      fileList.innerHTML = '';
      
      selectedFiles.forEach(function(file, index) {
        var div = document.createElement('div');
        div.className = 'file-item';
        div.innerHTML = '<span>' + file.name + ' (' + (file.size/1024).toFixed(1) + ' KB)</span>' +
          '<span class="remove-file" onclick="removeFile(' + index + ')">×</span>';
        fileList.appendChild(div);
      });
    }
    
    window.removeFile = function(index) {
      selectedFiles.splice(index, 1);
      renderFileList();
    };
    
    var dropZone = document.getElementById('dropZone');
    
    dropZone.addEventListener('dragover', function(e) {
      e.preventDefault();
      dropZone.classList.add('active');
    });
    
    dropZone.addEventListener('dragleave', function() {
      dropZone.classList.remove('active');
    });
    
    dropZone.addEventListener('drop', function(e) {
      e.preventDefault();
      dropZone.classList.remove('active');
      window.handleFiles(e.dataTransfer.files);
    });
    
    window.submitExpenseForm = function() {
      var date = document.getElementById('date').value;
      if (!date) {
        alert('Please select a date');
        document.getElementById('date').focus();
        return;
      }
      
      var category = document.getElementById('category').value;
      if (!category) {
        alert('Please select a category');
        document.getElementById('category').focus();
        return;
      }
      
      var subCategory = document.getElementById('subCategory').value;
      if (!subCategory) {
        alert('Please select a sub-category');
        document.getElementById('subCategory').focus();
        return;
      }
      
      if (selectedMonths.length === 0) {
        alert('Please select at least one month');
        return;
      }
      
      var amount = parseFloat(document.getElementById('amount').value);
      if (!amount || amount <= 0) {
        document.getElementById('amountError').style.display = 'block';
        document.getElementById('amount').focus();
        return;
      }
      
      var btn = document.getElementById('submitBtn');
      btn.disabled = true;
      btn.innerHTML = '<span class="loading"></span>Processing...';
      
      var attachments = [];
      var pendingFiles = selectedFiles.length;
      
      if (pendingFiles === 0) {
        submitData(attachments);
        return;
      }
      
      selectedFiles.forEach(function(file) {
        var reader = new FileReader();
        reader.onloadend = function() {
          attachments.push({
            name: file.name,
            type: file.type,
            data: reader.result.split(',')[1]
          });
          pendingFiles--;
          if (pendingFiles === 0) {
            submitData(attachments);
          }
        };
        reader.onerror = function() {
          pendingFiles--;
          if (pendingFiles === 0) {
            submitData(attachments);
          }
        };
        reader.readAsDataURL(file);
      });
      
      function submitData(attachments) {
        var formData = {
          date: date,
          category: category,
          subCategory: subCategory,
          months: selectedMonths,
          amount: amount,
          notes: document.getElementById('notes').value,
          attachments: attachments,
          userPassword: document.getElementById('userPassword').value
        };
        
        fetch(window.scriptUrl, {
          method: 'POST',
          headers: {
            'Content-Type': 'application/x-www-form-urlencoded',
          },
          body: 'action=submitExpense&data=' + encodeURIComponent(JSON.stringify(formData))
        })
        .then(function(response) { return response.json(); })
        .then(function(result) {
          if (result.success) {
            document.getElementById('successMessage').textContent = result.message;
            document.getElementById('successModal').style.display = 'flex';
          } else {
            alert('Error: ' + result.error);
            btn.disabled = false;
            btn.innerHTML = 'Submit Expense';
          }
        })
        .catch(function(error) {
          alert('Error submitting: ' + error);
          btn.disabled = false;
          btn.innerHTML = 'Submit Expense';
        });
      }
    };
    
    window.resetExpenseForm = function() {
      document.getElementById('expenseForm').reset();
      document.getElementById('subCategory').disabled = true;
      document.getElementById('subCategory').innerHTML = '<option value="">Select Category First</option>';
      
      selectedMonths = [];
      var buttons = document.querySelectorAll('.month-btn');
      for (var i = 0; i < buttons.length; i++) {
        buttons[i].classList.remove('selected');
      }
      document.getElementById('monthSummary').textContent = '0 months selected';
      
      selectedFiles = [];
      renderFileList();
      
      document.getElementById('successModal').style.display = 'none';
      document.getElementById('submitBtn').disabled = false;
      document.getElementById('submitBtn').innerHTML = 'Submit Expense';
    };
  })();
</script>
