<script>
  (function() {
    const MONTH_NAMES = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
    
    function convertYyMmmToMmmYy(monthStr) {
      if (!monthStr) return '-';
      if (/^[A-Za-z]{3}-\d{2}$/.test(monthStr)) return monthStr;
      if (/^\d{2}-[A-Za-z]{3}$/.test(monthStr)) {
        const parts = monthStr.split('-');
        return parts[1] + '-' + parts[0];
      }
      return monthStr;
    }
    
    function loadDashboardData() {
      google.script.run
        .withSuccessHandler(function(data) {
          renderDashboard(JSON.parse(data));
        })
        .withFailureHandler(function(error) {
          document.getElementById('matrixContainer').innerHTML = '<div style="text-align: center; padding: 40px; color: #999;">Error loading data</div>';
          document.getElementById('transactionsTable').innerHTML = '<div style="text-align: center; padding: 40px; color: #999;">Error loading data</div>';
        })
        .getDashboardData();
    }
    
    function renderDashboard(data) {
      if (data.error) {
        console.error('Server error:', data.error);
      }
      
      document.getElementById('beginningBalance').textContent = formatCurrency(data.beginningBalance);
      document.getElementById('totalRevenue').textContent = formatCurrency(data.revenueTotal);
      document.getElementById('totalExpenses').textContent = formatCurrency(data.expenseTotal);
      document.getElementById('currentBalance').textContent = formatCurrency(data.currentBalance);
      
      const net = parseFloat(data.currentBalance);
      document.getElementById('currentBalance').style.color = net >= 0 ? '#43e97b' : '#f5576c';
      
      renderMatrix(data.matrixData);
      renderTransactions(data.transactions);
    }
    
    function renderMatrix(matrixData) {
      if (!matrixData || !matrixData.months || matrixData.months.length === 0) {
        document.getElementById('matrixContainer').innerHTML = '<div style="text-align: center; padding: 40px; color: #999;">No matrix data available</div>';
        return;
      }
      
      const months = matrixData.months;
      const tenants = matrixData.tenants;
      const payments = matrixData.payments;
      
      let html = '<div class="matrix-fixed"><table class="matrix-table"><thead><tr><th>Tenant</th></tr></thead><tbody>';
      
      tenants.forEach(function(tenant) {
        html += '<tr><td>' + tenant + '</td></tr>';
      });
      
      html += '</tbody></table></div>';
      
      html += '<div class="matrix-scroll"><table class="matrix-table"><thead><tr>';
      
      months.forEach(function(month) {
        const displayMonth = convertYyMmmToMmmYy(month);
        html += '<th>' + displayMonth + '</th>';
      });
      
      html += '</tr></thead><tbody>';
      
      tenants.forEach(function(tenant) {
        html += '<tr>';
        months.forEach(function(month) {
          const key = tenant + '|' + month;
          const status = payments[key] || 'unpaid';
          const cellClass = status === 'paid' ? 'paid' : 'unpaid';
          const cellContent = status === 'paid' ? '✓' : '✗';
          
          html += '<td class="' + cellClass + '">' + cellContent + '</td>';
        });
        html += '</tr>';
      });
      
      html += '</tbody></table></div>';
      
      document.getElementById('matrixContainer').innerHTML = html;
    }
    
        function renderTransactions(transactions) {
      if (!transactions || transactions.length === 0) {
        document.getElementById('transactionsTable').innerHTML = '<div style="text-align: center; padding: 40px; color: #999;">No transactions found</div>';
        return;
      }
      
      let html = '<table><thead><tr>' +
        '<th>Type</th>' +
        '<th>Date</th>' +
        '<th>Category</th>' +
        '<th>SubCategory</th>' +
        '<th>Month</th>' +
        '<th>Tenant</th>' +
        '<th>Amount</th>' +
        '<th>Note</th>' +
        '<th>Attachment</th>' +
        '</tr></thead><tbody>';
      
      transactions.slice(0, 50).forEach(function(row) {
        // Column indices: 0=Timestamp, 1=Type, 2=Date, 3=Category, 4=SubCategory, 5=Month, 6=Tenant, 7=Amount, 8=Note, 9=Attachment
        const type = row[1];
        const date = row[2];
        const category = row[3];
        const subCategory = row[4];
        const monthRaw = row[5];
        const tenant = row[6];
        const amount = row[7];
        const note = row[8];
        const attachment = row[9];
        
        const isExpense = type === 'Expense';
        const amountClass = isExpense ? 'expense-amount' : 'revenue-amount';
        const badgeClass = isExpense ? 'badge-expense' : 'badge-revenue';
        
        // Format month to mmm-yy
        let monthStr = formatMonthToMmmYy(monthRaw);
        
        html += '<tr>';
        html += '<td><span class="badge ' + badgeClass + '">' + (type || '-') + '</span></td>';
        html += '<td>' + formatDate(date) + '</td>';
        html += '<td>' + (category || '-') + '</td>';
        html += '<td>' + (subCategory || '-') + '</td>';
        html += '<td>' + monthStr + '</td>';
        html += '<td>' + (tenant || '-') + '</td>';
        html += '<td class="amount ' + amountClass + '">' + formatAmount(amount) + '</td>';
        html += '<td>' + (note || '-') + '</td>';
        
        // Attachment as clickable link(s)
        if (attachment && attachment.toString().trim()) {
          const urls = attachment.toString().split(',').map(u => u.trim()).filter(u => u);
          if (urls.length > 0) {
            html += '<td>';
            urls.forEach(function(url, idx) {
              html += '<a href="' + url + '" target="_blank" class="attachment-link">Link ' + (idx + 1) + '</a>';
              if (idx < urls.length - 1) html += ', ';
            });
            html += '</td>';
          } else {
            html += '<td>-</td>';
          }
        } else {
          html += '<td>-</td>';
        }
        
        html += '</tr>';
      });
      
      html += '</tbody></table>';
      document.getElementById('transactionsTable').innerHTML = html;
    }
    
    function formatMonthToMmmYy(monthRaw) {
      if (!monthRaw) return '-';
      
      // If already in mmm-yy format
      if (typeof monthRaw === 'string' && /^[A-Za-z]{3}-\d{2}$/.test(monthRaw)) {
        return monthRaw;
      }
      
      // If in yy-mmm format, convert to mmm-yy
      if (typeof monthRaw === 'string' && /^\d{2}-[A-Za-z]{3}$/.test(monthRaw)) {
        const parts = monthRaw.split('-');
        return parts[1] + '-' + parts[0];
      }
      
      // If it's a Date object
      if (monthRaw instanceof Date) {
        const mm = MONTH_NAMES[monthRaw.getMonth()];
        const yy = String(monthRaw.getFullYear()).slice(-2);
        return mm + '-' + yy;
      }
      
      // If it's mm-yy format
      if (typeof monthRaw === 'string' && /^\d{2}-\d{2}$/.test(monthRaw)) {
        const parts = monthRaw.split('-');
        const mm = MONTH_NAMES[parseInt(parts[0]) - 1];
        const yy = parts[1];
        return mm + '-' + yy;
      }
      
      // If it's an ISO date string
      if (typeof monthRaw === 'string' && monthRaw.includes('T')) {
        const d = new Date(monthRaw);
        if (!isNaN(d.getTime())) {
          const mm = MONTH_NAMES[d.getMonth()];
          const yy = String(d.getFullYear()).slice(-2);
          return mm + '-' + yy;
        }
      }
      
      // If it's a string like "11/24/2026"
      if (typeof monthRaw === 'string' && monthRaw.includes('/')) {
        const parts = monthRaw.split('/');
        if (parts.length === 3) {
          const mm = MONTH_NAMES[parseInt(parts[0]) - 1];
          const yy = parts[2].slice(-2);
          return mm + '-' + yy;
        }
      }
      
      return String(monthRaw);
    }
    
    function formatCurrency(value) {
      const num = parseFloat(value);
      if (isNaN(num)) return 'EGP 0';
      if (num >= 1000) return 'EGP ' + (num/1000).toFixed(1) + 'K';
      return 'EGP ' + num.toFixed(0);
    }
    
    function formatAmount(value) {
      const num = parseFloat(value);
      if (isNaN(num)) return 'EGP 0';
      const absVal = Math.abs(num);
      if (absVal >= 1000) {
        const prefix = num < 0 ? '-' : '';
        return prefix + 'EGP ' + (absVal/1000).toFixed(1) + 'K';
      }
      const prefix = num < 0 ? '-' : '';
      return prefix + 'EGP ' + absVal.toFixed(0);
    }
    
    function formatDate(dateVal) {
      if (!dateVal) return '-';
      if (typeof dateVal === 'string' && dateVal.indexOf('/') > -1) {
        // Return short format for mobile
        const parts = dateVal.split('/');
        if (parts.length === 3) {
          return parts[0] + '/' + parts[1];
        }
        return dateVal;
      }
      
      try {
        const date = new Date(dateVal);
        if (isNaN(date.getTime())) return String(dateVal);
        
        const day = String(date.getDate()).padStart(2, '0');
        const month = String(date.getMonth() + 1).padStart(2, '0');
        
        return day + '/' + month;
      } catch(e) {
        return String(dateVal);
      }
    }
    
    loadDashboardData();
  })();
</script>
